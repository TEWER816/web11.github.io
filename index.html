API 参考

极致体验的核心能力

// 动画引擎
EIWKlib.animate.tween(el, {
x: 100,
opacity: 1,
scale: 1.2
}, { duration: 500 })
// 缓动函数
EIWKlib.animate.easing.easeOutBounce(t)
EIWKlib.animate.easing.bezier(
0.4, 0, 0.2, 1
)(t)
// DOM 操作
EIWKlib.dom.create('div', {
className: 'box',
style: { opacity: 0.5 }
}, 'content')
// 工具函数
EIWKlib.utils.debounce(fn, 300)
EIWKlib.utils.throttle(fn, 100)
EIWKlib.utils.pipe(fn1, fn2, fn3)
// 加载屏幕
EIWKlib.loadingScreen.show({
duration: 30000,
title: 'EIWKlib',
subtitle: '追求极致体验',
demos: [fn1, fn2],
demoInterval: 6500
})
.loadingScreen.hide()
.loadingScreen.cancel()
// 按空格键可跳过
// 物理模拟
EIWKlib.animate.physics.spring({
from: 0, to: 100,
stiffness: 100, damping: 10
})
.animate.physics.gravity({
from: 0, to: 80, bounce: 0.7
})
